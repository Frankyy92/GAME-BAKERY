async function loadAssets() {

    const bar = document.getElementById('loaderBar');

    const text = document.getElementById('loaderText');

    

    const assetList = Object.entries(ASSETS);

    

    for (let i = 0; i < assetList.length; i++) {

        const [name, url] = assetList[i];

        text.textContent = `Chargement: ${name}...`;

        bar.style.width = `${((i + 1) / assetList.length) * 100}%`;

        

        try {

            textures[name] = await PIXI.Assets.load(url);

        } catch(e) {

            console.warn(`Failed to load ${name}:`, e);

        }

    }

    

    // Découper les produits (grille 2x2)

    if (textures.products) {

        const base = textures.products.baseTexture;

        const pw = base.width / 2, ph = base.height / 2;

        const padding = 0.12; // 12% padding

        

        for (const [key, data] of Object.entries(PRODUCTS)) {

            const rect = new PIXI.Rectangle(

                data.col * pw + pw * padding,

                data.row * ph + ph * padding,

                pw * (1 - padding * 2),

                ph * (1 - padding * 2)

            );

            textures[`product_${key}`] = new PIXI.Texture(base, rect);

        }

    }

    

    // Découper les effets (4 lignes x 8 colonnes)

    if (textures.effects) {

        const base = textures.effects.baseTexture;

        const cols = 8, rows = 4;

        const fw = base.width / cols, fh = base.height / rows;

        

        textures.smokeFrames = [];

        textures.sparkleFrames = [];

        textures.coinFrames = [];

        

        for (let i = 0; i < cols; i++) {

            textures.smokeFrames.push(new PIXI.Texture(base, new PIXI.Rectangle(i * fw, 0, fw, fh)));

            textures.sparkleFrames.push(new PIXI.Texture(base, new PIXI.Rectangle(i * fw, fh, fw, fh)));

            textures.coinFrames.push(new PIXI.Texture(base, new PIXI.Rectangle(i * fw, fh * 2, fw, fh)));

        }

    }

    

    // Découper les clients (estimation 8 personnages en ligne)

    if (textures.customers) {

        const base = textures.customers.baseTexture;

        const numCustomers = 8;

        const cw = base.width / numCustomers;

        

        textures.customerSprites = [];

        for (let i = 0; i < numCustomers; i++) {

            textures.customerSprites.push(

                new PIXI.Texture(base, new PIXI.Rectangle(i * cw, 0, cw, base.height))

            );

        }

    }

    

    text.textContent = 'Prêt !';

    await delay(300);

}

    return zone;

function onProductClick(productId) {

    if (gameState.preparing && gameState.preparing !== productId) return;

    

    if (gameState.ready === productId) {

        serveCustomer(productId);

    } else if (!gameState.preparing) {

        startPreparation(productId);

    }

}


function startPreparation(productId) {

    gameState.preparing = productId;

    gameState.prepProgress = 0;

    

    // Update status

    const prepZone = gameContainer.getChildByName('prepZone');

    const status = prepZone.getChildByName('status');

    status.text = '⏳ Cuisson en cours...';

    status.style.fill = 0xFBBF24;

    

    // Highlight bouton

    const btn = productButtons.find(b => b.id === productId);

    if (btn) {

        btn.container.getChildByName('bg').tint = 0xFFF3CD;

    }

    

    // Ajouter fumée sur le décor

    spawnSmoke();

}


function completePreparation() {

    const productId = gameState.preparing;

    gameState.ready = productId;

    gameState.preparing = null;

    gameState.prepProgress = 0;

    

    // Update status

    const prepZone = gameContainer.getChildByName('prepZone');

    const status = prepZone.getChildByName('status');

    status.text = '✓ PRÊT ! Clique pour servir';

    status.style.fill = 0x22C55E;

    

    // Show badge

    const btn = productButtons.find(b => b.id === productId);

    if (btn) {

        btn.container.getChildByName('badge').visible = true;

        btn.container.getChildByName('check').visible = true;

    }

}


function serveCustomer(productId) {

    const idx = gameState.customers.findIndex(c => c.order === productId && !c.served);

    

    if (idx !== -1) {

        const customer = gameState.customers[idx];

        const data = PRODUCTS[productId];

        

        // Points

        const bonus = Math.floor(customer.patience / 4);

        const comboBonus = gameState.combo * 10;

        const total = data.points + bonus + comboBonus;

        

        gameState.score += total;

        gameState.served++;

        gameState.combo++;

        customer.served = true;

        

        // Effets

        spawnFloatingText(`+${total}`, customer.sprite.x, customer.sprite.y - 50, 0x22C55E);

        spawnSparkle(customer.sprite.x, customer.sprite.y - 30);

        spawnCoins(customer.sprite.x, customer.sprite.y);

        

        if (gameState.combo >= 3) {

            showComboText();

        }

        

        // Animation sortie client

        animateCustomerLeave(customer);

        

        updateUI();

    }

    

    // Reset bouton

    gameState.ready = null;

    const btn = productButtons.find(b => b.id === productId);

    if (btn) {

        btn.container.getChildByName('bg').tint = 0xFFFFFF;

        btn.container.getChildByName('badge').visible = false;

        btn.container.getChildByName('check').visible = false;

    }

    

    // Reset status

    const prepZone = gameContainer.getChildByName('prepZone');

    const status = prepZone.getChildByName('status');

    status.text = '👆 Sélectionne un produit';

    status.style.fill = 0xAAAAAA;

}

function spawnCustomer() {

    if (gameState.screen !== 'playing') return;

    

    const lvl = LEVELS[gameState.level];

    if (gameState.spawned >= lvl.customers) return;

    

    const activeCount = gameState.customers.filter(c => !c.served).length;

    if (activeCount >= lvl.maxQueue) {

        setTimeout(spawnCustomer, 500);

        return;

    }

    

    // Choisir produit aléatoire

    const prods = lvl.products;

    const order = prods[Math.floor(Math.random() * prods.length)];

    

    // Type visuel

    const type = Math.floor(Math.random() * (textures.customerSprites?.length || 1));

    

    // Position isométrique

    const queueIndex = activeCount;

    const startX = CONFIG.width + 80;

    const targetX = 280 - queueIndex * 50; // File d'attente isométrique

    const targetY = 350 + queueIndex * 25; // Plus bas = plus proche

    

    const customer = {

        id: Date.now(),

        order,

        patience: 100,

        served: false,

        type,

        sprite: null

    };

    

    // Créer sprite

    const sprite = createCustomerSprite(customer, type);

    sprite.x = startX;

    sprite.y = targetY;

    sprite.zIndex = Math.floor(targetY); // Z-index dynamique

    customer.sprite = sprite;

    customersContainer.addChild(sprite);

    

    gameState.customers.push(customer);

    gameState.spawned++;

    

    // Animation entrée

    animateCustomerEnter(sprite, targetX, targetY);

    

    // Prochain client

    setTimeout(spawnCustomer, lvl.speed);

}


function createCustomerSprite(customer, type) {

    const container = new PIXI.Container();

    

    // Sprite personnage

    const tex = textures.customerSprites?.[type] || null;

    if (tex) {

        const sprite = new PIXI.Sprite(tex);

        sprite.anchor.set(0.5, 1);

        sprite.scale.set(0.5);

        container.addChild(sprite);

    } else {

        // Fallback cercle

        const placeholder = new PIXI.Graphics();

        placeholder.beginFill(0x4A90D9);

        placeholder.drawCircle(0, -30, 25);

        placeholder.beginFill(0xFFCCB3);

        placeholder.drawCircle(0, -50, 15);

        container.addChild(placeholder);

    }

    

    // Bulle commande

    const bubble = new PIXI.Container();

    bubble.position.set(0, -100);

    

    const bubbleBg = new PIXI.Graphics();

    bubbleBg.beginFill(0xFFFFFF);

    bubbleBg.lineStyle(2, 0xE8A849);

    bubbleBg.drawRoundedRect(-28, -28, 56, 56, 12);

    bubble.addChild(bubbleBg);

    

    // Triangle bulle

    const tri = new PIXI.Graphics();

    tri.beginFill(0xFFFFFF);

    tri.lineStyle(2, 0xE8A849);

    tri.moveTo(-8, 26);

    tri.lineTo(0, 38);

    tri.lineTo(8, 26);

    bubble.addChild(tri);

    

    // Produit dans bulle

    const prodTex = textures[`product_${customer.order}`];

    if (prodTex) {

        const prodSprite = new PIXI.Sprite(prodTex);

        prodSprite.anchor.set(0.5);

        prodSprite.width = 40;

        prodSprite.height = 40;

        bubble.addChild(prodSprite);

    }

    

    container.addChild(bubble);

    container.bubble = bubble;

    

    // Barre patience

    const barBg = new PIXI.Graphics();

    barBg.beginFill(0x333333);

    barBg.drawRoundedRect(-25, 5, 50, 8, 4);

    container.addChild(barBg);

    

    const barFill = new PIXI.Graphics();

    barFill.name = 'patienceBar';

    container.addChild(barFill);

    

    return container;

}


function animateCustomerEnter(sprite, targetX, targetY) {

    const startX = sprite.x;

    const duration = 600;

    const startTime = Date.now();

    

    const animate = () => {

        const elapsed = Date.now() - startTime;

        const t = Math.min(elapsed / duration, 1);

        const ease = 1 - Math.pow(1 - t, 3);

        

        sprite.x = startX + (targetX - startX) * ease;

        sprite.y = targetY - Math.sin(t * Math.PI) * 15;

        

        if (t < 1) requestAnimationFrame(animate);

    };

    animate();

}


function animateCustomerLeave(customer) {

    const sprite = customer.sprite;

    const duration = 400;

    const startTime = Date.now();

    const startX = sprite.x;

    

    const animate = () => {

        const elapsed = Date.now() - startTime;

        const t = Math.min(elapsed / duration, 1);

        

        sprite.x = startX - 100 * t;

        sprite.alpha = 1 - t;

        sprite.scale.set(1 - t * 0.3);

        

        if (t < 1) {

            requestAnimationFrame(animate);

        } else {

            customersContainer.removeChild(sprite);

            repositionQueue();

        }

    };

    animate();

}


function repositionQueue() {

    const active = gameState.customers.filter(c => !c.served);

    active.forEach((c, i) => {

        const targetX = 280 - i * 50;

        const targetY = 350 + i * 25;

        

        if (c.sprite) {

            const startX = c.sprite.x, startY = c.sprite.y;

            const duration = 300;

            const startTime = Date.now();

            

            const animate = () => {

                const t = Math.min((Date.now() - startTime) / duration, 1);

                c.sprite.x = startX + (targetX - startX) * t;

                c.sprite.y = startY + (targetY - startY) * t;

                c.sprite.zIndex = Math.floor(c.sprite.y);

                if (t < 1) requestAnimationFrame(animate);

            };

            animate();

        }

    });

}

function gameLoop(delta) {

    if (gameState.screen !== 'playing') return;

    

    const lvl = LEVELS[gameState.level];

    

    // Update patience

    gameState.customers.forEach(c => {

        if (c.served) return;

        

        c.patience -= 0.12 * delta;

        

        // Update barre

        if (c.sprite) {

            const bar = c.sprite.getChildByName('patienceBar');

            if (bar) {

                bar.clear();

                const color = c.patience > 60 ? 0x22C55E : c.patience > 30 ? 0xFBBF24 : 0xEF4444;

                bar.beginFill(color);

                bar.drawRoundedRect(-25, 5, (c.patience / 100) * 50, 8, 4);

            }

            

            // Shake si impatient

            if (c.patience < 25 && c.sprite.bubble) {

                c.sprite.bubble.x = Math.sin(Date.now() / 50) * 3;

            }

        }

        

        // Client perdu

        if (c.patience <= 0) {

            c.served = true;

            gameState.missed++;

            gameState.combo = 0;

            

            spawnFloatingText('😠', c.sprite.x, c.sprite.y - 60, 0xEF4444);

            

            if (c.sprite) {

                const sprite = c.sprite;

                const duration = 500;

                const startTime = Date.now();

                

                const animate = () => {

                    const t = Math.min((Date.now() - startTime) / duration, 1);

                    sprite.alpha = 1 - t;

                    sprite.y -= 2;

                    if (t < 1) requestAnimationFrame(animate);

                    else {

                        customersContainer.removeChild(sprite);

                        repositionQueue();

                    }

                };

                animate();

            }

            

            updateUI();

        }

    });

    

    // Update préparation

    if (gameState.preparing) {

        const data = PRODUCTS[gameState.preparing];

        gameState.prepProgress += (delta * 16) / data.time;

        

        // Update cercle progress

        const btn = productButtons.find(b => b.id === gameState.preparing);

        if (btn) {

            const prog = btn.container.getChildByName('progress');

            prog.visible = true;

            prog.clear();

            prog.lineStyle(4, 0xF59E0B);

            prog.arc(0, -15, 35, -Math.PI/2, -Math.PI/2 + Math.PI * 2 * gameState.prepProgress);

        }

        

        if (gameState.prepProgress >= 1) {

            // Hide progress

            if (btn) btn.container.getChildByName('progress').visible = false;

            completePreparation();

        }

    }

    

    // Check fin niveau

    if (gameState.served + gameState.missed >= lvl.customers) {

        endGame();

    }

}


function updateUI() {

    const header = gameContainer.getChildByName('header');

    if (!header) return;

    

    const lvl = LEVELS[gameState.level];

    

    // Score

    const score = header.getChildByName('scoreLabel');

    if (score) score.text = gameState.score.toString();

    

    // Info clients

    const info = header.getChildByName('clientInfo');

    if (info) info.text = `${gameState.served}/${lvl.customers}`;

    

    // Barre progression

    const bar = header.getChildByName('progressBar');

    if (bar) {

        const progress = Math.min(gameState.score / lvl.goal, 1);

        bar.clear();

        bar.beginFill(progress >= 1 ? 0x22C55E : 0x60A5FA);

        bar.drawRoundedRect(15, 52, (CONFIG.width - 30) * progress, 6, 3);

    }

}

function endGame() {

    gameState.screen = 'result';

    

    const lvl = LEVELS[gameState.level];

    const stars = gameState.missed === 0 ? 3 : gameState.missed <= 2 ? 2 : gameState.missed <= 3 ? 1 : 0;

    const success = stars >= 1 && gameState.score >= lvl.goal;

    

    gameState.stars[gameState.level] = Math.max(gameState.stars[gameState.level], stars);

    

    const earned = Math.floor(gameState.score / 8) + stars * 15;

    gameState.coins += earned;

    

    setTimeout(() => showResult(success, stars, earned), 600);

}

    icon.position.set(CONFIG.width/2, 120);

    resultContainer.addChild(icon);

        resultContainer.addChild(val);

    });

    

    // Boutons

    const replayBtn = createActionButton('Rejouer 🔄', 0xF59E0B, () => startGame(gameState.level));

    replayBtn.position.set(CONFIG.width/2, 500);

    resultContainer.addChild(replayBtn);

    

    if (success && gameState.level < LEVELS.length - 1 && !LEVELS[gameState.level + 1].locked) {

        const nextBtn = createActionButton('Niveau suivant →', 0x22C55E, () => startGame(gameState.level + 1));

        nextBtn.position.set(CONFIG.width/2, 565);

        resultContainer.addChild(nextBtn);

    }

    

    const menuBtn = createActionButton('Menu', 0x6B7280, showMenu);

    menuBtn.position.set(CONFIG.width/2, success && gameState.level < 5 ? 630 : 565);

    resultContainer.addChild(menuBtn);

    

    app.stage.addChild(resultContainer);

}


function createActionButton(label, color, onClick) {

    const btn = new PIXI.Container();

    btn.eventMode = 'static';

    btn.cursor = 'pointer';

    

    const bg = new PIXI.Graphics();

    bg.beginFill(color);

    bg.drawRoundedRect(-150, -28, 300, 56, 15);

    btn.addChild(bg);

    

    const text = createText(label, { fontSize: 20, fill: 0xFFFFFF, fontWeight: 'bold' });

    text.anchor.set(0.5);

    btn.addChild(text);

    

    btn.on('pointerdown', () => animatePress(btn, onClick));

    btn.on('pointerover', () => btn.scale.set(1.03));

    btn.on('pointerout', () => btn.scale.set(1));

    

    return btn;

}

function createText(content, style) {

    const defaultStyle = {

        fontFamily: 'Segoe UI, system-ui, sans-serif',

        fontSize: 16,

        fill: 0x333333

    };

    return new PIXI.Text(content, { ...defaultStyle, ...style });

}


function animatePress(obj, callback) {

    obj.scale.set(0.95);

    setTimeout(() => {

        obj.scale.set(1);

        callback();

    }, 100);

}
